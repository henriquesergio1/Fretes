# ===== Estágio 1: Build =====
FROM node:18-alpine AS builder

WORKDIR /app

# Instala as dependências de build
RUN npm install esbuild tailwindcss papaparse

# Copia os arquivos de configuração e código-fonte
COPY . .

# Compila o CSS com Tailwind
# A flag --minify é adicionada para produção
RUN ./node_modules/.bin/tailwindcss -i ./tailwind-input.css -o ./dist/output.css --minify

# Compila e agrupa o TypeScript/React com esbuild
RUN ./node_modules/.bin/esbuild index.tsx --bundle --outfile=dist/bundle.js --minify --loader:.tsx=tsx

# Gera o arquivo index.html de produção
# Este HTML aponta para os arquivos compilados (JS e CSS) e não usa CDNs.
RUN echo '<!DOCTYPE html>\
<html lang="pt-BR">\
<head>\
<meta charset="UTF-8" />\
<meta name="viewport" content="width=device-width, initial-scale=1.0" />\
<title>Fretes</title>\
<link href="/dist/output.css" rel="stylesheet">\
<style>\
::-webkit-scrollbar{width:8px;height:8px}\
::-webkit-scrollbar-track{background:#1e293b}\
::-webkit-scrollbar-thumb{background:#475569;border-radius:10px}\
::-webkit-scrollbar-thumb:hover{background:#64748b}\
</style>\
</head>\
<body class="bg-slate-900 text-slate-100">\
<div id="root"></div>\
<script type="module" src="/dist/bundle.js"></script>\
</body>\
</html>' > ./dist/index.html

# ===== Estágio 2: Produção =====
FROM nginx:alpine

# Copia a configuração personalizada do Nginx
COPY nginx.conf.txt /etc/nginx/conf.d/default.conf

# Copia os arquivos estáticos compilados do estágio de build
COPY --from=builder /app/dist/ /usr/share/nginx/html/dist/
COPY --from=builder /app/dist/index.html /usr/share/nginx/html/index.html

EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]
