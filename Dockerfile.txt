# ===== Estágio 1: Build =====
FROM node:18-alpine AS builder

WORKDIR /app

# Instala as dependências de build
RUN npm install esbuild tailwindcss papaparse

# Copia os arquivos de configuração e código-fonte
COPY . .

# Compila o CSS com Tailwind
# A flag --minify é adicionada para produção
RUN ./node_modules/.bin/tailwindcss -i ./tailwind-input.css -o ./dist/output.css --minify

# Compila e agrupa o TypeScript/React com esbuild
RUN ./node_modules/.bin/esbuild index.tsx --bundle --outfile=dist/bundle.js --minify --loader:.tsx=tsx

# ===== Estágio 2: Produção =====
FROM nginx:alpine

# Copia a configuração personalizada do Nginx
COPY nginx.conf.txt /etc/nginx/conf.d/default.conf

# Copia os arquivos estáticos compilados do estágio de build
COPY --from=builder /app/dist/ /usr/share/nginx/html/dist/

# Copia o arquivo HTML de produção (que foi copiado para o builder no passo 'COPY . .')
COPY --from=builder /app/index.prod.html /usr/share/nginx/html/index.html

EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]
