
server {
    listen 80;
    server_name localhost;

    # Adiciona os tipos MIME padrão do Nginx
    include /etc/nginx/mime.types;

    root /usr/share/nginx/html;
    index index.html;

    # Desativa cache para o index.html para evitar configurações antigas
    location = /index.html {
        add_header Cache-Control "no-store, no-cache, must-revalidate";
        expires 0;
    }

    # Configuração do Proxy Reverso para a API
    # O navegador chama http://[IP]:8080/api/veiculos
    # O Nginx reescreve para http://fretes-api:3000/veiculos
    location /api/ {
        rewrite ^/api/(.*) /$1 break;
        proxy_pass http://fretes-api:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }

    # Configuração do Frontend (SPA)
    location / {
        # Tenta servir o arquivo solicitado diretamente.
        # Se falhar, redireciona para /index.html para o roteamento do React.
        try_files $uri $uri/ /index.html;
    }
}
